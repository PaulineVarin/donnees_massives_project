                                $regexMatch : { input : "$albums.genre", regex : /Rock/, options : "i" }
                            },

                        {
                            case : {
                                $regexMatch : { input : "$albums.genre", regex : /Metal/, options : "i" }
                            },
                            then : "Metal"
                        },

                        {
                            case : {
                                $regexMatch : { input : "$albums.genre", regex : /Country/, options : "i" }
